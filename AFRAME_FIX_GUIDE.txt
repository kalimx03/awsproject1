// src/main.jsx
import React from 'react'
import ReactDOM from 'react-dom/client'
import App from './App.jsx'
import ErrorBoundary from './components/shared/ErrorBoundary.jsx'
import './styles/main.css'

// =====================================================
// CRITICAL: A-Frame is loaded in index.html BEFORE this
// script, so AFRAME should be available as a global
// =====================================================
console.log('main.jsx initializing...')
console.log('AFRAME available:', typeof AFRAME !== 'undefined')

// Wait for AFRAME to be ready if it exists
function waitForAFrame(timeout = 3000) {
  return new Promise((resolve) => {
    if (typeof AFRAME !== 'undefined' && AFRAME.isComponentized) {
      console.log('✓ A-Frame is ready')
      resolve(true)
      return
    }
    
    let elapsed = 0
    const checkInterval = setInterval(() => {
      elapsed += 100
      
      if (typeof AFRAME !== 'undefined' && AFRAME.isComponentized) {
        console.log('✓ A-Frame loaded successfully')
        clearInterval(checkInterval)
        resolve(true)
        return
      }
      
      if (elapsed >= timeout) {
        console.warn('⚠ A-Frame timeout - continuing without it')
        clearInterval(checkInterval)
        resolve(false)
      }
    }, 100)
  })
}

// Initialize AWS Amplify (if configured)
async function initializeAmplify() {
  try {
    const { Amplify } = await import('aws-amplify')
    const awsExports = (await import('../aws-exports.js')).default
    
    if (awsExports && awsExports.Auth && awsExports.Auth.Cognito) {
      Amplify.configure(awsExports)
      console.log('✓ AWS Amplify configured')
      return true
    } else {
      console.log('⚠ AWS configuration incomplete, using default settings')
      return false
    }
  } catch (error) {
    console.warn('AWS Amplify not configured:', error.message)
    return false
  }
}

// Main initialization function
async function main() {
  try {
    // Hide loading screen
    const loadingScreen = document.getElementById('loadingScreen')
    
    // Wait for A-Frame to initialize
    console.log('Waiting for A-Frame...')
    await waitForAFrame(3000)
    
    // Initialize AWS (optional)
    console.log('Initializing AWS...')
    await initializeAmplify()
    
    // Render React app
    console.log('Rendering React app...')
    const root = ReactDOM.createRoot(document.getElementById('root'))
    
    root.render(
      <React.StrictMode>
        <ErrorBoundary>
          <App />
        </ErrorBoundary>
      </React.StrictMode>
    )
    
    // Hide loading screen after a short delay to ensure render
    setTimeout(() => {
      if (loadingScreen) {
        loadingScreen.classList.add('hidden')
      }
    }, 100)
    
    console.log('✓ Application initialized successfully')
  } catch (error) {
    console.error('Fatal error during initialization:', error)
    
    // Show error message
    const loadingScreen = document.getElementById('loadingScreen')
    if (loadingScreen) {
      loadingScreen.innerHTML = `
        <div style="text-align: center; color: #ef4444;">
          <div style="font-size: 3rem; margin-bottom: 1rem;">⚠️</div>
          <div style="font-size: 1.5rem; font-weight: bold; margin-bottom: 0.5rem;">Initialization Error</div>
          <div style="color: rgba(255,255,255,0.7); margin-bottom: 1rem; font-size: 0.875rem;">
            ${error.message || 'An unknown error occurred'}
          </div>
          <button onclick="location.reload()" style="
            padding: 0.75rem 1.5rem;
            background: #7c3aed;
            color: white;
            border: none;
            border-radius: 0.5rem;
            cursor: pointer;
          ">Retry</button>
        </div>
      `
    }
  }
}

// Start the application when DOM is ready
if (document.readyState === 'loading') {
  document.addEventListener('DOMContentLoaded', main)
} else {
  main()
}

===============================================
TROUBLESHOOTING:
===============================================

If you still see white screen:
1. Check browser console (F12) for errors
2. Verify internet connection (CDN files need to load)
3. Check Network tab - are CDN scripts loading?
4. Try: Hard refresh (Ctrl+Shift+R or Cmd+Shift+R)

If AFRAME is still not defined:
1. Check Network tab - aframe.min.js must fully load
2. Check HTML source - verify scripts are in <head>
3. Verify no other scripts run before A-Frame loads

If CSP errors appear:
1. Check the CSP meta tag in index.html
2. Verify script-src includes all needed domains
3. Don't use 'unsafe-eval' with this setup
